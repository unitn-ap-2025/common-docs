1//! # Planet module
2//! This module provides common definitions for planets and their associated types
3//! that need be used by a group to construct its own planet.
4//! The [Planet] struct is the **main component**: an instance of it represents the
5//! actual planet and contains all the logic and state (see [PlanetState]) needed to work as one, in fact
6//! this is what the orchestrator will interact with.
7//!
8//! You can instantiate a new planet by calling the [Planet::new] constructor method and passing
9//! valid construction parameters to it (look into its documentation to learn more).
10//!
11//! One of the construction parameters is a group-defined struct that implements the [PlanetAI] trait,
12//! which defines several methods for handling messages coming from the orchestrator and the explorers. This is
13//! the core of each group's planet implementation, as it defines the planet *behaviour*, that is
14//! how a planet "reacts" to the possible events or requests.
15//!
16//! ## Examples
17//! Intended usage (for planet definition, by groups):
18//!
19//! ```
20//! use crossbeam_channel::{Sender, Receiver};
21//! use common_game::components::planet::{Planet, PlanetAI, PlanetState, PlanetType};
22//! use common_game::components::resource::{Combinator, Generator};
23//! use common_game::components::rocket::Rocket;
24//! use common_game::protocols::messages;
25//!
26//! // Group-defined AI struct
27//! struct AI { /* your AI state here */ };
28//!
29//! impl PlanetAI for AI {
30//!     fn handle_orchestrator_msg(
31//!         &mut self,
32//!         state: &mut PlanetState,
33//!         generator: &Generator,
34//!         combinator: &Combinator,
35//!         msg: messages::OrchestratorToPlanet
36//!     ) -> Option<messages::PlanetToOrchestrator> {
37//!         // your handler code here...
38//!         None
39//!     }
40//!
41//!     fn handle_explorer_msg(
42//!         &mut self,
43//!         state: &mut PlanetState,
44//!         generator: &Generator,
45//!         combinator: &Combinator,
46//!         msg: messages::ExplorerToPlanet
47//!     ) -> Option<messages::PlanetToExplorer> {
48//!         // your handler code here...
49//!         None
50//!     }
51//!
52//!     fn handle_asteroid(
53//!         &mut self,
54//!         state: &mut PlanetState,
55//!         generator: &Generator,
56//!         combinator: &Combinator,
57//!     ) -> Option<Rocket> {
58//!         // your handler code here...
59//!         None
60//!     }
61//!
62//!     fn start(&mut self, state: &PlanetState) { /* startup code */ }
63//!     fn stop(&mut self, state: &PlanetState) { /* stop code */ }
64//! }
65//!
66//! // This is the group's "export" function. It will be called by
67//! // the orchestrator to spawn your planet.
68//! pub fn create_planet(
69//!     rx_orchestrator: Receiver<messages::OrchestratorToPlanet>,
70//!     tx_orchestrator: Sender<messages::PlanetToOrchestrator>,
71//!     rx_explorer: Receiver<messages::ExplorerToPlanet>,
72//! ) -> Planet {
73//!     let id = 1;
74//!     let ai = AI {};
75//!     let gen_rules = vec![/* your recipes */];
76//!     let comb_rules = vec![/* your recipes */];
77//!
78//!     // Construct the planet and return it
79//!     Planet::new(
80//!         id,
81//!         PlanetType::A,
82//!         Box::new(ai),
83//!         gen_rules,
84//!         comb_rules,
85//!         (rx_orchestrator, tx_orchestrator),
86//!         rx_explorer,
87//!     ).unwrap() // Don't call .unwrap()! You should do error checking instead.
88//! }
89//! ```

102/// The trait that defines the behaviour of a planet.
103///
104/// Structs implementing this trait are intended to be passed to the
105/// [Planet] constructor, so that the handlers can be invoked by the planet
106/// internal logic when certain messages are received on any of the planet channels.
107///
108/// The handlers can alter the planet state by accessing the
109/// `state` parameter, which is passed to the methods as a mutable borrow.
110/// A response can be sent by returning an optional message of the correct type,
111/// that will be forwarded to the associated channel passed on planet construction.
112pub trait PlanetAI: Send {
113    /// Handler for messages received by the orchestrator (receiving
114    /// end of the [OrchestratorToPlanet] channel).
115    /// The following messages will **not** invoke this handler:
116    /// - [OrchestratorToPlanet::StartPlanetAI] (see [PlanetAI::start])
117    /// - [OrchestratorToPlanet::StopPlanetAI] (see [PlanetAI::stop])
118    /// - [OrchestratorToPlanet::Asteroid] (see [PlanetAI::handle_asteroid])
119    /// - [OrchestratorToPlanet::IncomingExplorerRequest], as this will be handled automatically by the planet
120    /// - [OrchestratorToPlanet::OutgoingExplorerRequest] (same as previous one)
121    ///
122    /// Check [PlanetAI] docs for general meaning of the parameters and return type.
123    fn handle_orchestrator_msg(
124        &mut self,
125        state: &mut PlanetState,
126        generator: &Generator,
127        combinator: &Combinator,
128        msg: OrchestratorToPlanet,
129    ) -> Option<PlanetToOrchestrator>;
130
131    /// Handler for **all** messages received by an explorer (receiving
132    /// end of the [ExplorerToPlanet] channel).
133    ///
134    /// Check [PlanetAI] docs for general meaning of the parameters and return type.
135    fn handle_explorer_msg(
136        &mut self,
137        state: &mut PlanetState,
138        generator: &Generator,
139        combinator: &Combinator,
140        msg: ExplorerToPlanet,
141    ) -> Option<PlanetToExplorer>;
142
143    /// This handler will be invoked when a [OrchestratorToPlanet::Asteroid]
144    /// message is received. It's important to handle *Asteroid* messages
145    /// correctly, as this will the determine the planet survival.
146    ///
147    /// # Returns
148    /// In order to survive, an owned [Rocket] **must** be returned from this method;
149    /// if `None` is returned instead, the planet will (or *should*) be **destroyed** by the orchestrator
150    fn handle_asteroid(
151        &mut self,
152        state: &mut PlanetState,
153        generator: &Generator,
154        combinator: &Combinator,
155    ) -> Option<Rocket>;
156
157    /// This method will be invoked when a [OrchestratorToPlanet::StartPlanetAI]
158    /// is received, but **only if** the planet is currently in a *stopped* state.
159    ///
160    /// Start messages received when planet is already running are **ignored**.
161    fn start(&mut self, state: &PlanetState);
162
163    /// This method will be invoked when a [OrchestratorToPlanet::StopPlanetAI]
164    /// is received, but **only if** the planet is currently in a *running* state.
165    ///
166    /// Stop messages received when planet is already stopped are **ignored**.
167    fn stop(&mut self, state: &PlanetState);
168}

170/// Contains planet rules constraints (see [PlanetType]).
171pub struct PlanetConstraints {
172    n_energy_cells: usize,
173    unbounded_gen_rules: bool,
174    can_have_rocket: bool,
175    n_comb_rules: usize,

178/// Planet types definitions, intended to be passed
179/// to the planet constructor. Identifies the planet rules constraints,
180/// with each type having its own.
181#[derive(Debug, Clone, Copy)]
182pub enum PlanetType {
183    A,
184    B,
185    C,
186    D,
187}
188
189impl PlanetType {
190    const N_ENERGY_CELLS: usize = 5;
191    const N_RESOURCE_COMB_RULES: usize = 6;
192
193    /// Returns a tuple with the constraints associated to the planet type,
194    /// as described in the project specifications.
195    pub fn constraints(&self) -> PlanetConstraints {
196        match self {
197            PlanetType::A => PlanetConstraints {
198                n_energy_cells: Self::N_ENERGY_CELLS,
199                unbounded_gen_rules: false,
200                can_have_rocket: true,
201                n_comb_rules: 0,
202            },
203            PlanetType::B => PlanetConstraints {
204                n_energy_cells: 1,
205                unbounded_gen_rules: true,
206                can_have_rocket: false,
207                n_comb_rules: 1,
208            },
209            PlanetType::C => PlanetConstraints {
210                n_energy_cells: 1,
211                unbounded_gen_rules: false,
212                can_have_rocket: true,
213                n_comb_rules: Self::N_RESOURCE_COMB_RULES,
214            },
215            PlanetType::D => PlanetConstraints {
216                n_energy_cells: Self::N_ENERGY_CELLS,
217                unbounded_gen_rules: true,
218                can_have_rocket: false,
219                n_comb_rules: 0,
220            },
221        }
222    }
223}

225/// This struct is a representation of the internal state
226/// of the planet. Through its public methods, it gives access to the all resources
227/// of the planet:
228/// - A vec of [EnergyCell].
229/// - An optional [Rocket], that can be built accordingly to the planet type.
230/// - [Generator] for generating basic resources.
231/// - [Combinator] for combining basic resources into complex ones.
232pub struct PlanetState {
233    id: u32,
234    energy_cells: Vec<EnergyCell>,
235    rocket: Option<Rocket>,
236    can_have_rocket: bool,
237}
238
239impl PlanetState {
240    /// Returns the planet id.
241    pub fn id(&self) -> u32 {
242        self.id
243    }
244
245    /// Indexed getter accessor for the [EnergyCell] vec.
246    ///
247    /// # Returns
248    /// An immutable borrow of the *i-th* energy cell.
249    ///
250    /// # Panics
251    /// This method will panic if the index `i` is out of bounds.
252    /// Always check the number of energy cells available with [PlanetState::cells_count].
253    pub fn cell(&self, i: usize) -> &EnergyCell {
254        &self.energy_cells[i]
255    }
256
257    /// Indexed *mutable* getter accessor for the [EnergyCell] vec.
258    ///
259    /// # Returns
260    /// An mutable borrow of the *i-th* energy cell.
261    ///
262    /// # Panics
263    /// This method will panic if the index `i` is out of bounds.
264    /// Always check the number of energy cells available with [PlanetState::cells_count].
265    pub fn cell_mut(&mut self, i: usize) -> &mut EnergyCell {
266        &mut self.energy_cells[i]
267    }
268
269    /// Returns the number of energy cells owned by
270    /// the planet. This is the actual size of the internal
271    /// vec containing the cells.
272    pub fn cells_count(&self) -> usize {
273        self.energy_cells.len()
274    }
275
276    /// Returns an *immutable* iterator over the energy cells owned by the planet.
277    pub fn cells_iter(&self) -> Iter<'_, EnergyCell> {
278        self.energy_cells.iter()
279    }
280
281    /// Returns a *mutable* iterator over the energy cells owned by the planet.
282    pub fn cells_iter_mut(&mut self) -> IterMut<'_, EnergyCell> {
283        self.energy_cells.iter_mut()
284    }
285
286    /// Charges the first empty (discharged) cell.
287    /// Returns an optional [Sunray] if there's no cell to charge.
288    pub fn charge_cell(&mut self, sunray: Sunray) -> Option<Sunray> {
289        match self.empty_cell() {
290            None => Some(sunray),
291            Some((cell, _)) => {
292                cell.charge(sunray);
293                None
294            }
295        }
296    }
297
298    /// Returns a tuple containing a *mutable* borrow of the first empty (discharged) cell
299    /// and its index, or `None` if there isn't any.
300    pub fn empty_cell(&mut self) -> Option<(&mut EnergyCell, usize)> {
301        let idx = self.energy_cells.iter().position(|cell| !cell.is_charged());
302        idx.map(|i| (&mut self.energy_cells[i], i))
303    }
304
305    /// Returns a tuple containing a *mutable* borrow of the first full (charged) cell
306    /// and its index, or `None` if there isn't any.
307    pub fn full_cell(&mut self) -> Option<(&mut EnergyCell, usize)> {
308        let idx = self.energy_cells.iter().position(|cell| cell.is_charged());
309        idx.map(|i| (&mut self.energy_cells[i], i))
310    }
311
312    /// Returns `true` if the planet can have a rocket.
313    pub fn can_have_rocket(&self) -> bool {
314        self.can_have_rocket
315    }
316
317    /// Returns `true` if the planet has a rocket built and ready to launch.
318    pub fn has_rocket(&self) -> bool {
319        self.rocket.is_some()
320    }
321
322    /// Takes the rocket out of the planet state (if there is one), leaving
323    /// `None` in its place.
324    pub fn take_rocket(&mut self) -> Option<Rocket> {
325        self.rocket.take()
326    }
327
328    /// Constructs a rocket using the *i-th* [EnergyCell] of the planet and stores it
329    /// inside the planet, taking ownership of it.
330    ///
331    /// # Panics
332    /// This method will panic if the index `i` is out of bounds.
333    /// Always check the number of energy cells available with [PlanetState::cells_count].
334    ///
335    /// # Errors
336    /// Returns an error if:
337    /// - The planet type prohibits the storing of rockets.
338    /// - The planet already has a rocket built.
339    /// - The energy cell is not charged
340    pub fn build_rocket(&mut self, i: usize) -> Result<(), String> {
341        if !self.can_have_rocket {
342            Err("This planet type can't have rockets.".to_string())
343        } else if self.has_rocket() {
344            Err("This planet already has a rocket.".to_string())
345        } else {
346            let energy_cell = self.cell_mut(i);
347            Rocket::new(energy_cell).map(|rocket| {
348                self.rocket = Some(rocket);
349            })
350        }
351    }
352
353    /// Returns a *dummy* clone of this state.
354    pub fn to_dummy(&self) -> DummyPlanetState {
355        DummyPlanetState {
356            energy_cells: self
357                .energy_cells
358                .iter()
359                .map(|cell| cell.is_charged())
360                .collect(),
361            charged_cells_count: self
362                .energy_cells
363                .iter()
364                .filter(|cell| cell.is_charged())
365                .count(),
366            has_rocket: self.has_rocket(),
367        }
368    }
369}

382/// Main, top-level planet definition. This type is built on top of
383/// [PlanetState], [PlanetType] and [PlanetAI], through composition.
384///
385/// It needs to be constructed by each group as it represents the actual planet
386/// and contains the base logic that runs the AI. Also, this is what should be
387/// returned to the orchestrator.
388///
389/// See module-level docs for more general info.
390pub struct Planet {
391    state: PlanetState,
392    planet_type: PlanetType,
393    pub ai: Box<dyn PlanetAI>,
394    generator: Generator,
395    combinator: Combinator,
396
397    from_orchestrator: Receiver<OrchestratorToPlanet>,
398    to_orchestrator: Sender<PlanetToOrchestrator>,
399    from_explorers: Receiver<ExplorerToPlanet>,
400    to_explorers: HashMap<u32, Sender<PlanetToExplorer>>,
401}
402
403impl Planet {
404    const ORCH_DISCONNECT_ERR: &str = "Orchestrator disconnected.";
405
406    /// Constructor for the [Planet] type.
407    ///
408    /// # Errors
409    /// Returns an error if the construction parameters are *invalid* (they violate the `planet_type` constraints).
410    ///
411    /// # Arguments
412    /// - `id` - The identifier to assign to the planet.
413    /// - `planet_type` - Type of the planet. Constraints the rules of the planet.
414    /// - `ai` - A group-defined struct implementing the [PlanetAI] trait.
415    /// - `gen_rules` - A vec of [BasicResourceType] containing the basic resources the planet will be able to generate.
416    /// - `comb_rules` - A vec of [ComplexResourceType] containing the complex resources the planet will be able to make.
417    /// - `orchestrator_channels` - A pair containing the receiver and sender half
418    ///   of the channels [OrchestratorToPlanet] and [PlanetToOrchestrator].
419    /// - `explorers_receiver` - The receiver half of the [ExplorerToPlanet] channel
420    ///   where all explorers send messages to this planet (when they're visiting it).
421    pub fn new(
422        id: u32,
423        planet_type: PlanetType,
424        ai: Box<dyn PlanetAI>,
425        gen_rules: Vec<BasicResourceType>,
426        comb_rules: Vec<ComplexResourceType>,
427        orchestrator_channels: (Receiver<OrchestratorToPlanet>, Sender<PlanetToOrchestrator>),
428        explorers_receiver: Receiver<ExplorerToPlanet>,
429    ) -> Result<Planet, String> {
430        let PlanetConstraints {
431            n_energy_cells,
432            unbounded_gen_rules,
433            can_have_rocket,
434            n_comb_rules,
435        } = planet_type.constraints();
436        let (from_orchestrator, to_orchestrator) = orchestrator_channels;
437
438        if gen_rules.is_empty() {
439            Err("gen_rules is empty".to_string())
440        } else if !unbounded_gen_rules && gen_rules.len() > 1 {
441            Err(format!(
442                "Too many generation rules (Planet type {:?} is limited to 1)",
443                planet_type
444            ))
445        } else if comb_rules.len() > n_comb_rules {
446            Err(format!(
447                "Too many combination rules (Planet type {:?} is limited to {})",
448                planet_type, n_comb_rules
449            ))
450        } else {
451            let mut generator = Generator::new();
452            let mut combinator = Combinator::new();
453
454            // add gen and comb rules to the planet generator and combinator
455            for r in gen_rules {
456                let _ = generator.add(r);
457            }
458            for r in comb_rules {
459                let _ = combinator.add(r);
460            }
461
462            Ok(Planet {
463                state: PlanetState {
464                    id,
465                    energy_cells: (0..n_energy_cells).map(|_| EnergyCell::new()).collect(),
466                    can_have_rocket,
467                    rocket: None,
468                },
469                planet_type,
470                ai,
471                generator,
472                combinator,
473                from_orchestrator,
474                to_orchestrator,
475                from_explorers: explorers_receiver,
476                to_explorers: HashMap::new(),
477            })
478        }
479    }
480
481    /// Starts the planet in a *stopped* state, waiting for a [OrchestratorToPlanet::StartPlanetAI] message,
482    /// then invokes [PlanetAI::start] and runs the main message polling loop.
483    /// See [PlanetAI] docs to know more about when message handlers are invoked and how the planet reacts
484    /// to the different messages.
485    ///
486    /// This method is *blocking* and should be called by the orchestrator in a separate thread.
487    /// It returns with an [Ok] when the planet has been **destroyed** (killed).
488    ///
489    /// # Errors
490    /// If the orchestrator disconnects from the channel, this will return an [Err].
491    pub fn run(&mut self) -> Result<(), String> {
492        // run the planet stopped by default
493        // and wait for a StartPlanetAI message
494        let kill = self.wait_for_start()?;
495        if kill {
496            return Ok(());
497        }
498
499        self.ai.start(&self.state);
500
501        loop {
502            select_biased! {
503                // wait for orchestrator message (prioritized operation)
504                recv(self.from_orchestrator) -> msg => match msg {
505                    Ok(OrchestratorToPlanet::StartPlanetAI) => {}
506
507                    Ok(OrchestratorToPlanet::StopPlanetAI) => {
508                        self.to_orchestrator
509                            .send(PlanetToOrchestrator::StopPlanetAIResult {
510                                planet_id: self.id(),
511                            })
512                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
513                        self.ai.stop(&self.state);
514
515                        let kill = self.wait_for_start()?; // blocking wait
516                        if kill { return Ok(()) }
517
518                        // restart AI
519                        self.ai.start(&self.state)
520                    }
521
522                    Ok(OrchestratorToPlanet::KillPlanet) => {
523                        self.to_orchestrator
524                            .send(PlanetToOrchestrator::KillPlanetResult { planet_id: self.id() })
525                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
526
527                        return Ok(())
528                    }
529
530                    Ok(OrchestratorToPlanet::Asteroid(_)) => {
531                        let rocket =
532                            self.ai
533                                .handle_asteroid(&mut self.state, &self.generator, &self.combinator);
534
535                        self.to_orchestrator
536                            .send(PlanetToOrchestrator::AsteroidAck {
537                                planet_id: self.id(),
538                                rocket
539                            })
540                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
541                    }
542
543                    Ok(OrchestratorToPlanet::IncomingExplorerRequest {
544                        explorer_id,
545                        new_mpsc_sender,
546                    }) => {
547                        self.to_explorers.insert(explorer_id, new_mpsc_sender); // add new explorer channel
548
549                        // send ack back to orchestrator
550                        self.to_orchestrator
551                            .send(PlanetToOrchestrator::IncomingExplorerResponse {
552                                planet_id: self.id(),
553                                res: Ok(()),
554                            })
555                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
556                    }
557
558                    Ok(OrchestratorToPlanet::OutgoingExplorerRequest { explorer_id }) => {
559                        self.to_explorers.remove(&explorer_id); // remove outgoing explorer channel
560
561                        // send ack back to orchestrator
562                        self.to_orchestrator
563                            .send(PlanetToOrchestrator::OutgoingExplorerResponse {
564                                planet_id: self.id(),
565                                res: Ok(()),
566                            })
567                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
568                    }
569
570                    // default case: relay to generic handler
571                    Ok(msg) => {
572                        self.ai
573                            .handle_orchestrator_msg(
574                                &mut self.state,
575                                &self.generator,
576                                &self.combinator,
577                                msg,
578                            )
579                            .map(|response| self.to_orchestrator.send(response))
580                            .transpose()
581                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
582                    }
583
584                    Err(_) => {
585                        return Err(Self::ORCH_DISCONNECT_ERR.to_string())
586                    }
587                },
588
589                // wait for explorer message (ignore disconnections)
590                recv(self.from_explorers) -> msg => if let Ok(msg) = msg {
591                    let explorer_id = msg.explorer_id();
592
593                    // if requesting explorer is currently
594                    // on the planet respond to it
595                    if let Some(to_explorer) = self.to_explorers.get(&explorer_id)
596                        && let Some(response) = self.ai.handle_explorer_msg(
597                            &mut self.state,
598                            &self.generator,
599                            &self.combinator,
600                            msg,
601                        )
602                    {
603                        to_explorer
604                            .send(response)
605                            .map_err(|_| format!("Explorer {} disconnected.", explorer_id))?;
606                    }
607                }
608            }
609        }
610    }

612    // private helper function that blocks until
613    // a StartPlanetAI message is received
614    fn wait_for_start(&self) -> Result<bool, String> {
615        loop {
616            select_biased! {
617                // orch messages
618                recv(self.from_orchestrator) -> msg => match msg {
619                    // if `Start` is received, return false
620                    Ok(OrchestratorToPlanet::StartPlanetAI) => {
621                        self.to_orchestrator
622                            .send(PlanetToOrchestrator::StartPlanetAIResult {
623                                planet_id: self.id(),
624                            })
625                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
626
627                        return Ok(false);
628                    }
629                    // if `Kill` is received, return true
630                    Ok(OrchestratorToPlanet::KillPlanet) => {
631                        self.to_orchestrator
632                            .send(PlanetToOrchestrator::KillPlanetResult { planet_id: self.id() })
633                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?;
634
635                        return Ok(true)
636                    }
637                    // every other message we respond with `Stopped`
638                    Ok(_) => {
639                        self.to_orchestrator
640                            .send(PlanetToOrchestrator::Stopped {
641                                planet_id: self.id(),
642                            })
643                            .map_err(|_| Self::ORCH_DISCONNECT_ERR.to_string())?
644                    }
645
646                    Err(_) => return Err(Self::ORCH_DISCONNECT_ERR.to_string()),
647                },
648
649                // explorers messages
650                recv(self.from_explorers) -> msg => if let Ok(msg) = msg &&
651                    let Some(to_explorer) = self.to_explorers.get(&msg.explorer_id())
652                {
653                    let _ = to_explorer.send(PlanetToExplorer::Stopped);
654                }
655            }
656        }
657    }

926
927        // Spawn thread

929            let _ = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {

933                    Err(err) => {

935                    }
