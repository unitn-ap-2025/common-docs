9/// Who is sending / receiving this event.

14    Orchestrator,

20/// Log level / channel.

52/// Simple key–value payload: string → string.
53pub type Payload = BTreeMap<String, String>;
54
55#[derive(Debug, Clone)]
56pub struct LogEvent {
57    pub timestamp_unix: i64,
58    pub sender_type: ActorType,
59    pub sender_id: u64,
60    pub receiver_type: ActorType,
61    pub receiver_id: String,
62    pub event_type: EventType,
63    pub channel: Channel,
64    pub payload: Payload,
65}
66
67impl LogEvent {
68    /// Helper: create an event with the current UNIX timestamp.
69    pub fn new(
70        sender_type: ActorType,
71        sender_id: impl Into<u64>,
72        receiver_type: ActorType,
73        receiver_id: impl Into<String>,
74        event_type: EventType,
75        channel: Channel,
76        payload: Payload,
77    ) -> Self {
78        let now = SystemTime::now()
79            .duration_since(UNIX_EPOCH)
80            .unwrap_or_default()
81            .as_secs() as i64;
82
83        Self {
84            timestamp_unix: now,
85            sender_type,
86            sender_id: sender_id.into(),
87            receiver_type,
88            receiver_id: receiver_id.into(),
89            event_type,
90            channel,
91            payload,
92        }
93    }
94
95    pub fn id_from_str(s: &str) -> u64 {
96        let mut hasher = DefaultHasher::new();
97        s.hash(&mut hasher);
98        hasher.finish()
99    }
100
101    /// Emit this event using the `log` crate.
102    ///
103    /// If no logger is initialized by the final binary,
104    /// this will just be a no-op (which is fine for a library).
105    pub fn emit(&self) {
106        use Channel::*;
107
108        match self.channel {
109            Error => log::error!("{:?}", self),
110            Warning => log::warn!("{:?}", self),
111            Info => log::info!("{:?}", self),
112            Debug => log::debug!("{:?}", self),
113            Trace => log::trace!("{:?}", self),
114        }
115    }
116}
117
118impl fmt::Display for LogEvent {
119    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
120        write!(
121            f,
122            "LogEvent {{ ts: {}, sender: {:?}#{}, receiver: {:?}/{}, event: {:?}, channel: {:?}, payload: {:?} }}",
123            self.timestamp_unix,
124            self.sender_type,
125            self.sender_id,
126            self.receiver_type,
127            self.receiver_id,
128            self.event_type,
129            self.channel,
130            self.payload
131        )
132    }
