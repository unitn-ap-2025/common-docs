17/// Messages sent by the `Orchestrator` to a `Planet`.
18#[derive(Debug)]
19pub enum OrchestratorToPlanet {
20    /// This variant is used to send a [Sunray] to a planet
21    Sunray(Sunray),
22    /// This variant is used to send an [Asteroid] to a planet
23    Asteroid(Asteroid),
24    /// This variant is used to start a Planet Ai and restart it if it is stopped
25    StartPlanetAI,
26    /// This variant is used to pause the planet Ai
27    StopPlanetAI,
28    /// This variant is used to kill (or destroy) the planet
29    KillPlanet,
30    /// This variant is used to obtain a Planet Internal State
31    InternalStateRequest,
32    /// This variant is used to send the new [Sender] of the incoming explorer, see the sequence diagram for more info
33    IncomingExplorerRequest {
34        explorer_id: u32,
35        new_mpsc_sender: Sender<PlanetToExplorer>,
36    },
37    /// This variant is used to notify the planet to drop the [Sender] of the outgoing explorer
38    OutgoingExplorerRequest { explorer_id: u32 },
39}

41/// Messages sent by a `Planet` to the `Orchestrator`.
42#[derive(Debug)]
43pub enum PlanetToOrchestrator {
44    /// This variant is used to acknowledge the obtained [Sunray]
45    SunrayAck { planet_id: u32 },
46    /// This variant is used to acknowledge the obtained [Asteroid] and notify the orchestrator
47    /// if the planet has been destroyed or not.
48    AsteroidAck {
49        planet_id: u32,
50        rocket: Option<Rocket>,
51    },
52    /// This variant is used to acknowledge the start of the Planet Ai
53    StartPlanetAIResult { planet_id: u32 },
54    /// This variant is used to acknowledge the stop of the Planet Ai
55    StopPlanetAIResult { planet_id: u32 },
56    /// This variant is used to acknowledge the killing of a planet
57    KillPlanetResult { planet_id: u32 },
58    /// This variant is used to send back the Planet State
59    InternalStateResponse {
60        planet_id: u32,
61        planet_state: DummyPlanetState,
62    },
63    /// This variant is used to acknowledge the incoming explorer
64    /// Encapsulates a [Result] with a possible [Err] String representing an error occurred
65    IncomingExplorerResponse {
66        planet_id: u32,
67        res: Result<(), String>,
68    },
69    /// This variant is used to acknowledge the outgoing explorer
70    /// Encapsulates a [Result] with a possible [Err] String representing an error occurred
71    OutgoingExplorerResponse {
72        planet_id: u32,
73        res: Result<(), String>,
74    },
75    /// This variant is used by planets that are currently in a *stopped* state
76    /// to acknowledge any message coming from the Orchestrator (except for [OrchestratorToPlanet::StartPlanetAI])
77    Stopped { planet_id: u32 },
78}
79
80impl PlanetToOrchestrator {
81    /// Helper method to extract the `planet_id` field from any message variant
82    /// without needing to match a specific one.
83    pub fn planet_id(&self) -> u32 {
84        match self {
85            PlanetToOrchestrator::SunrayAck { planet_id, .. } => *planet_id,
86            PlanetToOrchestrator::AsteroidAck { planet_id, .. } => *planet_id,
87            PlanetToOrchestrator::StartPlanetAIResult { planet_id, .. } => *planet_id,
88            PlanetToOrchestrator::StopPlanetAIResult { planet_id, .. } => *planet_id,
89            PlanetToOrchestrator::KillPlanetResult { planet_id, .. } => *planet_id,
90            PlanetToOrchestrator::InternalStateResponse { planet_id, .. } => *planet_id,
91            PlanetToOrchestrator::IncomingExplorerResponse { planet_id, .. } => *planet_id,
92            PlanetToOrchestrator::OutgoingExplorerResponse { planet_id, .. } => *planet_id,
93            PlanetToOrchestrator::Stopped { planet_id, .. } => *planet_id,
94        }
95    }
96}

98/// Messages sent by the `Orchestrator` to an `Explorer`.
99#[derive(Debug)]
100pub enum OrchestratorToExplorer {
101    /// This variant is used to start the Explorer AI
102    StartExplorerAI,
103    /// This variant is used to reset the Explorer AI
104    ResetExplorerAI,
105    /// This variant is used to kill the Explorer AI
106    KillExplorerAI,
107    /// This variant is used to send a [Sender] to the new planet
108    MoveToPlanet {
109        sender_to_new_planet: Option<Sender<ExplorerToPlanet>>,
110    }, //none if explorer asks to move to a non-adjacent planet,
111    /// This variant is used to ask the ID of the Planet in which the Explorer is currently located
112    CurrentPlanetRequest,
113    /// This variant is used to enforce the Explorer to ask the supported Resources on the Planet
114    SupportedResourceRequest,
115    /// This variant is used to enforce the Explorer to ask the supported Combinations on the Planet
116    SupportedCombinationRequest,
117    /// This variant is used to enforce the Explorer to ask the Planet to Generate a [BasicResource]
118    GenerateResourceRequest { to_generate: BasicResourceType },
119    /// This variant is used to enforce the Explorer to ask the Planet to Generate a [ComplexResource] using the [ComplexResourceRequest]
120    CombineResourceRequest(ComplexResourceRequest),
121    /// This variant is used to ask the content of the Explorer Bag
122    BagContentRequest,
123    /// This variant is used to send to the Explorer its neighbors' IDs
124    NeighborsResponse { neighbors: Vec<u32> },
125}

127/// Messages sent by an `Explorer` to the `Orchestrator`.
128#[derive(Debug)]
129pub enum ExplorerToOrchestrator<T> {
130    /// Acknowledge of [OrchestratorToExplorer::StartExplorerAI]
131    StartExplorerAIResult { explorer_id: u32 },
132    /// Acknowledge of [OrchestratorToExplorer::KillExplorerAI]
133    KillExplorerAIResult { explorer_id: u32 },
134    /// Acknowledge of [OrchestratorToExplorer::ResetExplorerAI]
135    ResetExplorerAIResult { explorer_id: u32 },
136    /// Acknowledge of [OrchestratorToExplorer::MoveToPlanet]
137    MovedToPlanetResult { explorer_id: u32 },
138    /// This variant is used to send the ID of the current planet on which the Explorer is located
139    CurrentPlanetResult { explorer_id: u32, planet_id: u32 },
140    /// This variant is used to send the list of the available [BasicResourceType] in the planet
141    SupportedResourceResult {
142        explorer_id: u32,
143        supported_resources: HashSet<BasicResourceType>,
144    },
145    /// This variant is used to send the list of the available [ComplexResourceType] in the planet
146    SupportedCombinationResult {
147        explorer_id: u32,
148        combination_list: HashSet<ComplexResourceType>,
149    },
150    /// This variant holds a [Result] for the Orchestrator to know if the requested [BasicResource] has been generated
151    GenerateResourceResponse {
152        explorer_id: u32,
153        generated: Result<(), ()>,
154    },
155    /// This variant holds a [Result] for the Orchestrator to know if the requested [ComplexResource] has been generated
156    CombineResourceResponse {
157        explorer_id: u32,
158        generated: Result<(), ()>,
159    },
160    /// This message is for passing around the bag content and has been implemented with a generic type to let the group the freedom to implement the methods on it
161    ///
162    /// ## Example
163    /// ```ignore
164    /// use std::collections::HashMap;
165    /// use common_game::components::resource::{ComplexResourceType, BasicResourceType};
166    /// use common_game::protocols::messages::ExplorerToOrchestrator;
167    ///
168    /// pub struct DummyBag {
169    ///     pub complex: HashMap<ComplexResourceType, u32>,
170    ///     pub basic: HashMap<BasicResourceType, u32>,
171    /// }
172    ///
173    /// let message = ExplorerToOrchestrator::BagContentResponse {
174    ///     explorer_id: 1,
175    ///     bag_content: DummyBag {
176    ///         complex: HashMap::new(),
177    ///         basic: HashMap::new(),
178    ///     }
179    /// };
180    /// ```
181    ///
182    BagContentResponse { explorer_id: u32, bag_content: T },
183    /// This variant asks the Orchestrator for the list of neighbors Planets to travel to
184    NeighborsRequest {
185        explorer_id: u32,
186        current_planet_id: u32,
187    },
188    /// This variant asks the Orchestrator to be sent to the specified Planet
189    TravelToPlanetRequest {
190        explorer_id: u32,
191        current_planet_id: u32,
192        dst_planet_id: u32,
193    },
194}

196impl<T> ExplorerToOrchestrator<T> {
197    /// Helper method to extract the `explorer_id` field from any message variant
198    /// without needing to match a specific one.
199    pub fn explorer_id(&self) -> u32 {
200        match self {
201            Self::StartExplorerAIResult { explorer_id, .. } => *explorer_id,
202            Self::KillExplorerAIResult { explorer_id, .. } => *explorer_id,
203            Self::ResetExplorerAIResult { explorer_id, .. } => *explorer_id,
204            Self::MovedToPlanetResult { explorer_id, .. } => *explorer_id,
205            Self::CurrentPlanetResult { explorer_id, .. } => *explorer_id,
206            Self::SupportedResourceResult { explorer_id, .. } => *explorer_id,
207            Self::SupportedCombinationResult { explorer_id, .. } => *explorer_id,
208            Self::GenerateResourceResponse { explorer_id, .. } => *explorer_id,
209            Self::CombineResourceResponse { explorer_id, .. } => *explorer_id,
210            Self::BagContentResponse { explorer_id, .. } => *explorer_id,
211            Self::NeighborsRequest { explorer_id, .. } => *explorer_id,
212            Self::TravelToPlanetRequest { explorer_id, .. } => *explorer_id,
213        }
214    }
215}

217/// Messages sent by an `Explorer` to a `Planet`.
218#[derive(Debug)]
219pub enum ExplorerToPlanet {
220    /// This variant is used to ask the Planet for the available [BasicResourceType]
221    SupportedResourceRequest { explorer_id: u32 },
222    /// This variant is used to ask the Planet for the available [ComplexResourceType]
223    SupportedCombinationRequest { explorer_id: u32 },
224    /// This variant is used to ask the Planet to generate a [BasicResource]
225    GenerateResourceRequest {
226        explorer_id: u32,
227        resource: BasicResourceType,
228    },

230    CombineResourceRequest {
231        explorer_id: u32,
232        msg: ComplexResourceRequest,
233    },
234    /// This variant is used to ask the Planet for the available energy_cells number
235    AvailableEnergyCellRequest { explorer_id: u32 },
236}

252/// Messages sent by a `Planet` to an `Explorer`.
253#[derive(Debug)]
254pub enum PlanetToExplorer {
255    /// This variant is used to send the available [BasicResourceType] list to the Explorer
256    SupportedResourceResponse {
257        resource_list: HashSet<BasicResourceType>,
258    },
259    /// This variant is used to send the available [ComplexResourceType] list to the Explorer
260    SupportedCombinationResponse {
261        combination_list: HashSet<ComplexResourceType>,
262    },
263    /// This variant is used to send the Optional [BasicResource] generated or [None] in case of errors
264    GenerateResourceResponse { resource: Option<BasicResource> },
265    /// This variant is used to send the [ComplexResource] generated
266    /// It contains a [Result] giving back the [ComplexResource] in case of success
267    /// and a triplet containing an error string and the two [GenericResource] provided by the Explorer
268    CombineResourceResponse {
269        complex_response: Result<ComplexResource, (String, GenericResource, GenericResource)>,
270    },
271    /// This variant is used to send the number of available energy cells to the Explorer
272    AvailableEnergyCellResponse { available_cells: u32 },
273    /// This variant is used by planets that are currently in a *stopped* state
274    /// to acknowledge any message coming from an explorer
275    Stopped,
