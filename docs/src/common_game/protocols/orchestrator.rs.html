21// Messages that the Orchestrator can send to a Planet.
22// Start/Stop AI have been wrapped in dedicated structs (StartPlanetAiMsg / StopPlanetAiMsg)
23// to constrain and clarify their signatures.
24
25// Using a struct instead of a bare enum variant argument gives us type-safety and
26// the possibility to extend this message later without changing the enum shape.
27#[allow(unused)]
28pub trait OrchestratorTrait {
29    // • Initializes planets (planet definitions are loaded from the galaxy initialization file).
30    // Returns a type implementing GalaxyTrait, representing the logical galaxy abstraction.
31    fn initialize_galaxy(&mut self, path: &str) -> impl GalaxyTrait;
32
33    // • Constructs planets and explorers.
34
35    // For now, we use a string from the initialization file to initialize every planet.
36    // This matches the PDF’s notion that planet configuration is file-driven.
37    fn make_planet<T: PlanetAI>(&self, init_sting: String) -> Planet<T>;
38
39    // Creates a new explorer.
40    // In the PDF, explorers are also constructed and managed by the orchestrator.
41    fn make_explorer() -> Explorer;
42
43    // • Distributes all channels and starts the game.
44    // The orchestrator is responsible for wiring up all entities and communication links.
45    fn start_game(path: &str) -> Self;
46
47    // • Is the only Sunray constructor.
48    fn create_sunray() -> Sunray;
49
50    // • Is the only Asteroid constructor.
51    fn create_asteroid() -> Asteroid;
52
53    // Functions for Orchestrator → Planet.
54    // These methods conceptually wrap sending an OrchestratorToPlanet message on the correct channel.
55
56    fn send_sunray<T, E>(&self, s: Sunray, planet_id: u32) -> Result<T, E>;
57
58    fn send_asteroid<T, E>(&self, a: Asteroid, planet_id: u32) -> Result<T, E>;
59
60    fn start_planet_ai<T, E>(&self, msg: StartPlanetAiMsg, planet_id: u32) -> Result<T, E>;
61
62    fn stop_planet_ai<T, E>(&self, msg: StopPlanetAiMsg, planet_id: u32) -> Result<T, E>;
63
64    // Functions for Orchestrator → Explorer.
65    // These methods conceptually wrap sending an OrchestratorToExplorer message
66    // on the explorer’s command channel.
